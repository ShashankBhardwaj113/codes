        //fill your code here
        getMonths = function () {
            var table = document.getElementById('rateTable');
            table.insertAdjacentHTML('beforebegin', '<h2>Rate table of monthly income</h2>');

            var arrMonths = ['April', 'May', 'June', 'July', 'August', 'September', 'October', 'November', 'December'];
            var arrValues = ['8', '6', '7.2', '4.6', '6.9', '8.5', '9.4', '5.1', '7.5'];
            for (var i = 0; i < 9; i++) {
                var tBodyRef = document.getElementById('rateTable').getElementsByTagName('tbody')[0];
                var newRow = tBodyRef.insertRow();
                var newCell1 = newRow.insertCell();
                var newCell2 = newRow.insertCell();
                var newText1 = document.createTextNode(arrMonths[i]);
                newCell1.appendChild(newText1);
                var newText2 = document.createTextNode(arrValues[i]);
                newCell2.appendChild(newText2);
            }

            document.getElementById('button').style.display = "none";
        }
    </script>
